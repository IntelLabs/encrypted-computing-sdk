183 template <typename INSTRUCTION>
184 Graph<INSTRUCTION> Graph<INSTRUCTION>::createGraph(std::vector<INSTRUCTION *> instructions)
185 {
186  Graph<INSTRUCTION> graph;
187  graph.network = TNodeEDatNet<Node<INSTRUCTION>, Edge>::New();
188 
189  int node_id = 0;
190  for (const auto &instruction : instructions)
191  {
192  int operation_ID = 0;
193  Node<INSTRUCTION> test(instruction->Name() + "_" + std::to_string(node_id), node_id, OPERATION, instruction);
194  graph.network->AddNode(node_id, test);
195  operation_ID = node_id;
196  graph.node_ID_Map[instruction->Name()] = std::vector<int>({ node_id });
197  node_id++;
198 
199  auto op_node = graph.network->GetNDat(operation_ID);
200  std::ostringstream oss;
201  oss << instruction->Name() << "(";
202  for (int x = 0; x < instruction->numInputOperands(); x++)
203  {
204  auto retrievedID = graph.node_ID_Map.find(instruction->getInputOperand(x).location());
205  int input_ID = 0;
206  if (retrievedID == graph.node_ID_Map.end())
207  {
208  Node<INSTRUCTION> input(instruction->getInputOperand(x).location(), node_id,
209  instruction->getInputOperand(x).immediate() ? IMMEDIATE : REGISTER_ADDRESS);
210  graph.network->AddNode(node_id, input);
211  input_ID = node_id;
212  graph.node_ID_Map[instruction->getInputOperand(x).location()] = std::vector<int>({ input_ID });
213  node_id++;
214  }
215  else
216  {
217  input_ID = retrievedID->second.back();
218  }
219 
220  graph.network->AddEdge(input_ID, operation_ID);
221 
222  auto node = graph.network->GetNDat(input_ID);
223  oss << node.output_string;
224  if (x < instruction->numInputOperands() - 1)
225  {
226  oss << ",";
227  }
228  }
229  oss << ")";
230  op_node.output_string = oss.str();
231  for (int x = 0; x < instruction->numOutputOperands(); x++)
232  {
233  auto retrievedID = graph.node_ID_Map.find(instruction->getOutputOperand(x).location());
234  int output_ID = 0;
235  if (retrievedID == graph.node_ID_Map.end())
236  {
237  Node<INSTRUCTION> input(instruction->getOutputOperand(x).location(), node_id, REGISTER_ADDRESS);
238  graph.network->AddNode(node_id, input);
239  output_ID = node_id;
240  graph.node_ID_Map[instruction->getOutputOperand(x).location()] = std::vector<int>({ output_ID });
241  node_id++;
242  }
243  else
244  {
245  Node<INSTRUCTION> input(instruction->getOutputOperand(x).location(), node_id, REGISTER_ADDRESS);
246  graph.network->AddNode(node_id, input);
247  output_ID = node_id;
248  retrievedID->second.push_back(node_id);
249  //node_ID_Map[instruction->getOutputOperand(x).location()]=std::vector<int>({output_ID});
250  node_id++;
251  }
252 
253  graph.network->AddEdge(operation_ID, output_ID);
254  }
255  }
256 
257  TIntStrH outputs;
258  return graph;
259 }

479 {

481  BFS.DoBfs(start_node_id, out_direction, in_direction, -1, TInt::Mx);
482  Network<INSTRUCTION> tree = Network<INSTRUCTION>::New();
483  BFS.NIdDistH.SortByDat();
484 
485  for (int i = 0; i < BFS.NIdDistH.Len(); i++)
486  {
487  const int NId = BFS.NIdDistH.GetKey(i);
488  const int Dist = BFS.NIdDistH[i];
489  auto NI = graph->GetNI(NId);
490  if (!tree->IsNode(NId))
491  {
492  auto dataNode = graph->GetNI(NId);
493  tree->AddNode(dataNode);
494  }
495  }
496 
497  for (int i = 0; i < BFS.NIdDistH.Len(); i++)
498  {
499  const int NId = BFS.NIdDistH.GetKey(i);
500  const int Dist = BFS.NIdDistH[i];
501  auto NI = graph->GetNI(NId);
502 
503  if (in_direction)
504  {
505  for (int e = 0; e < NI.GetInDeg(); e++)
506  {
507  const int Prev = NI.GetInNId(e);
508  // std::cout << "prev: " << Prev;
