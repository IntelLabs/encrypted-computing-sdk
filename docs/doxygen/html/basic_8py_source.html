31 def _mixed_to_pisa_ops(ops: Iterable[PIsaOp | list[PIsaOp] | HighOp]) -> list[PIsaOp]:
32  """Helper to process mixed list of op types to PIsaOp only"""
33 
34  def helper(op) -> list[PIsaOp]:
35  if isinstance(op, PIsaOp):
36  return [op]
37  if isinstance(op, list):
38  if not all(isinstance(elem, PIsaOp) for elem in op):
39  raise ValueError("Not all elements in list are pisa ops")
40  return op
41  return op.to_pisa()
42 
43  ops_pisa_clusters: Iterable[list[PIsaOp | HighOp]] = map(helper, ops)
44  # Flattens the list returned
45  return [pisa_op for pisa_ops in ops_pisa_clusters for pisa_op in pisa_ops]

60  def to_pisa(self) -> list[PIsaOp]:
61  """Return the p-isa equivalent of an Add"""
62  if self.input0.parts == self.input1.parts:
63  return [
64  self.op(self.context.label, *expand_io, rns)
65  for expand_io, rns in expand_ios(
66  self.context, self.output, self.input0, self.input1
67  )
68  ]
69 
70  # Not the same number of parts
71  first, second = (
72  (self.input0, self.input1)
73  if self.input0.parts < self.input1.parts
74  else (self.input1, self.input0)
75  )
76 
77  ls: list[PIsaOp] = []
78  for unit, q in it.product(
79  range(self.context.units), range(self.input0.start_rns, self.input0.rns)
80  ):
81  ls.extend(
82  self.op(
83  self.context.label,
84  self.output(part, q, unit),
85  first(part, q, unit),
86  second(0, q, unit),
87  q,
88  )
89  for part in range(first.parts)
90  )
91  ls.extend(
92  pisa_op.Copy(
93  self.context.label,
94  self.output(part, q, unit),
95  second(part, q, unit),
96  )
97  for part in range(first.parts, second.parts)
98  )
99  return ls

117 def convolution_indices(input0: Polys, input1: Polys) -> list[InIdxs]:
118  """Helper gives convolution of parts indices"""
119  # start_* is the deg + 1 of a polynomial (the vector)
120  idxs: list[InIdxs] = [[] for _ in range(input0.parts + input1.parts - 1)]
121  for t in it.product(
