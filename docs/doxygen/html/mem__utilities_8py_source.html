6 def computePriority(variable, replacement_policy):
7  """
8  Computes the priority for reusing the location of a specified variable.
9 
10  The priority is determined based on the replacement policy. The smaller the priority value,
11  the higher the priority for reuse. Tuples are used for staged comparisons.
12 
13  Args:
14  variable (Variable): The variable for which to compute the priority.
15  replacement_policy (str): The policy to use for determining priority. Must be one of
16  `Constants.REPLACEMENT_POLICIES`.
17 
18  Returns:
19  tuple: A tuple representing the priority for reusing the variable's location.
20  """
21  retval = (float("-inf"), ) # Default: highest priority if no variable
22  if variable:
23  # Register in use
24  # last_x_access = variable.last_x_access.bundle * Constants.MAX_BUNDLE_SIZE + variable.last_x_access.cycle \
25  last_x_access = variable.last_x_access if variable.last_x_access \
26  else CycleType(0, 0)
27  if replacement_policy == Constants.REPLACEMENT_POLICY_FTBU:
28  if variable.accessed_by_xinsts:
29  # Priority by
30  retval = (-variable.accessed_by_xinsts[0].index, # Largest (furthest) accessing instruction
31  *last_x_access, # Oldest accessed cycle (oldest == smallest)
32  len(variable.accessed_by_xinsts)) # How many more uses this variable has
33  elif replacement_policy == Constants.REPLACEMENT_POLICY_LRU:
34  # Priority by oldest accessed cycle (oldest == smallest)
35  retval = (*last_x_access, )
36  else:
37  raise ValueError(f'`replacement_policy`: invalid value "{replacement_policy}". Expected value in {REPLACEMENT_POLICIES}.')
38  return retval

40 def flushRegisterBank(register_bank,
41  current_cycle: CycleType,
42  replacement_policy,
43  live_var_names = None,
44  pct: float = 0.5):
45  """
46  Cleans up a register bank by removing variables assigned to registers.
47 
48  The function attempts to free up to pct * 100% of registers. Only non-dirty registers
49  that do not contain live variables are cleaned up. Dummy variables are considered live.
50 
51  Args:
52  register_bank (RegisterBank):
53  The register bank to clean up.
54  current_cycle (CycleType):
55  The current cycle to consider for readiness.
56  replacement_policy (str):
57  The policy to use for determining which variables to replace.
58  Must be one of `Constants.REPLACEMENT_POLICIES`.
59  live_var_names (set or list, optional):
60  A collection of variable names that are not available
61  for replacement. Defaults to None.
62  pct (float, optional):
63  The fraction of the register bank to clean up. Defaults to 0.5.
64  """
65  local_heap = PriorityQueue()
66  occupied_count: int = 0
67  for idx, reg in enumerate(register_bank):
68  # Traverse the registers in the bank and put occupied, non-dirty ones
69  # in a heap where priority is based on replacement_policy
70  v = reg.contained_variable
71  if v is not None:
72  occupied_count += 1
73  if not reg.register_dirty \
74  and (v.name and v.name not in live_var_names) \
75  and current_cycle >= v.cycle_ready:
76  # Variable can be cleared from the register if needed
77  priority = computePriority(v, replacement_policy)
78  local_heap.push(priority, reg, (idx, ))
79 
80  # Clean up registers until we reach the specified pct occupancy or we have
81  # no registers left that can be cleaned up
82  while local_heap \
83  and occupied_count / register_bank.register_count > pct:
84  _, reg = local_heap.pop()
85  reg.allocateVariable(None)
86  occupied_count -= 1

105 
106  Raises:
107  ValueError: If the replacement policy is invalid.
108 
109  Returns:
110  int: The index of the first empty location found in `vars_lst`, or the index of the suggested
111  location to replace if `vars_lst` is full and a replacement policy was specified. Returns -1
112  if no suitable location is found.
113  """
114  if replacement_policy and replacement_policy not in Constants.REPLACEMENT_POLICIES:
115  raise ValueError(('`replacement_policy`: invalid value "{}". '
116  'Expected value in {} or None.').format(replacement_policy,
117  Constants.REPLACEMENT_POLICIES))
118 
119  retval = -1
120  priority = (float("inf"), float("inf"), float("inf"))
121  for idx, v in enumerate(vars_lst):
122  if not v:
123  retval = idx
124  break # Found an empty spot
125  elif replacement_policy \
