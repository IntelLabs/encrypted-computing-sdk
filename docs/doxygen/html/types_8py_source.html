51  @classmethod
52  def from_polys(cls, poly: "Polys", *, mode: str | None = None) -> "Polys":
53  """Class method for creating a specific range of polys based on desired mode"""
54  copy = Polys(**vars(poly))
55  match mode:
56  case "drop_last_rns":
57  copy.rns -= 1
58  return cls(**vars(copy))
59  case "last_rns":

63  copy.rns = 1
64  return cls(**vars(copy))
65  case "last_part":

67  return cls(**vars(copy))

162  @classmethod
163  def from_string(cls, line: str):
164  """Construct context from a string"""
165  scheme, poly_order, key_rns, current_rns, *optionals = line.split()
166  optional_dict = OptionsDictParser.parse(optionals)
167  int_poly_order = int(poly_order)
168  if (
169  int_poly_order < MIN_POLY_SIZE
170  or int_poly_order > MAX_POLY_SIZE
171  or not math.log2(int_poly_order).is_integer()
172  ):
173  raise ValueError(
174  f"Poly order `{int_poly_order}` must be power of two >= {MIN_POLY_SIZE} and < {MAX_POLY_SIZE}"
175  )
176 
177  int_key_rns = int(key_rns)
178  int_current_rns = int(current_rns)
179  int_max_rns = int_key_rns - 1
180 
181  if int_key_rns <= int_current_rns:
182  raise ValueError(
183  f"Current RNS must be less than Key RNS: current_rns={current_rns}, key_rns={key_rns}"
184  )
185 
186  return cls(
187  scheme=scheme.upper(),
188  poly_order=int_poly_order,
189  key_rns=int_key_rns,
190  current_rns=int_current_rns,
191  max_rns=int_max_rns,
192  **optional_dict,
193  )
