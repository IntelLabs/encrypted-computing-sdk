21 def execute_process(cmd: list[str], data_in=None):
22  """Helper function for executing processes. stdout and stderr are always
23  captured. NOTE: subprocess.run will fail silently with a non-zero exit code
24  so always check the returncode"""
25  return run(
26  list(map(str, cmd)),
27  input=data_in,
28  capture_output=True,
29  check=False,
30  encoding="utf-8",
31  )
