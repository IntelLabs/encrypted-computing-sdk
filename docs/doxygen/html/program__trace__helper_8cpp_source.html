100 std::shared_ptr<pisa::poly::PolyProgram> PolynomialProgramHelper::parseCSV(const std::string &filename, bool ignore_header)
101 {
102  try
103  {
104  auto new_poly_program = pisa::poly::PolyProgram::create();
105 
106  std::ifstream file(filename);
107  if (!file.is_open())
108  {
109  throw std::runtime_error("File not found: " + filename);
110  }
111 
112  std::string current_line;
113  if (ignore_header)
114  std::getline(file, current_line);
115 
116  while (std::getline(file, current_line))
117  {
118  std::vector<std::string> components;
119  std::stringstream current_line_ss(current_line);
120 
121  std::string component;
122  while (std::getline(current_line_ss, component, ','))
123  {
124  components.push_back(trim(component));
125  }
126  new_poly_program->addOperation(parseInstruction(components, new_poly_program));
127  }
