25 std::vector<std::string> generateMemFile(const graph::Graph<pisa::PISAInstruction> &graph, int max_rns_terms)
26 {
27  const auto inputs = graph.getInputNodes(true, false);
28  const auto outputs = graph.getOutputNodes();
29 
30  int counter = 0;
31  std::vector<std::string> memory_file = {
32  "dload, ntt_auxiliary_table, " + std::to_string(counter++),
33  "dload, ntt_routing_table, " + std::to_string(counter++),
34  "dload, intt_auxiliary_table, " + std::to_string(counter++),
35  "dload, intt_routing_table, " + std::to_string(counter++)
36  };
37 
38  // Get twid/ones iterations
39  int high_rns_iters = 1 + ((max_rns_terms - 1) / 64);
40 
41  //Add preamble strings
42  std::generate_n(std::back_inserter(memory_file), 8 * high_rns_iters, [&counter]() {
43  return "dload, twid, " + std::to_string(counter++);
44  });
45  std::generate_n(std::back_inserter(memory_file), high_rns_iters, [&counter]() {
46  return "dload, ones, " + std::to_string(counter++);
47  });
48 
49  //Add inputs
50  std::map<std::string, int> hbm_address_map;
51  std::transform(inputs.begin(), inputs.end(), std::back_inserter(memory_file),
52  [&counter, &hbm_address_map](const auto &x) {
53  if (hbm_address_map.count(x.GetDat().label) == 0)
54  {
55  hbm_address_map[x.GetDat().label] = counter++;
56  }
57  return "dload, poly, " + std::to_string(hbm_address_map[x.GetDat().label]) + ", " + x.GetDat().label;
58  });
59 
60  //Add outputs
61  std::transform(outputs.begin(), outputs.end(), std::back_inserter(memory_file),
62  [&counter, &hbm_address_map](const auto &x) {
63  //dstore, output_0_0_0, 73
64  if (hbm_address_map.count(x.GetDat().label) == 0)
65  {
66  hbm_address_map[x.GetDat().label] = counter++;
67  }
68  return "dstore, " + x.GetDat().label + ", " + std::to_string(hbm_address_map[x.GetDat().label]);
69  });
70 
71  return memory_file;
72 }
