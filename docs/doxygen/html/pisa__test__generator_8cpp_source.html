188 void PisaTestGenerator::convertPolyRnsChunkToPolyRnsHelper(json &input_json)
189 {
190  std::map<std::string, std::vector<std::pair<std::string, int>>> collections;
191  for (auto input : input_json.items())
192  {
193  auto label = input.key();
194  int block_divider = label.find_last_of('_');
195  auto block_removed = label.substr(0, block_divider);
196  int block_number = std::stoi(label.substr(block_divider + 1));
197 
198  // This creates my ordered vectors
199  //std::cout << "Adding to collection: " << block_removed << " , " << label << " , " << block_number << std::endl;
200  collections[block_removed].push_back(std::pair<std::string, int>(label, block_number));
201  }
202  for (auto &collection : collections)
203  {
204  //Sort vector based on block number
205  std::sort(collection.second.begin(), collection.second.end(), [](const std::pair<std::string, int> &a, const std::pair<std::string, int> &b) { return a.second < b.second; });
206 
207  for (auto chunk : collection.second)
208  {
209  for (auto block_value : input_json[chunk.first].items())
210  {
211  //std::cout << "Block V"
212  input_json[collection.first][input_json[collection.first].size()] = block_value.value();
213  };
214  }
215  //Remove block item from json once data is transferred
216  int items_to_remove = collection.second.size();
217  for (int x = 0; x < items_to_remove; x++)
