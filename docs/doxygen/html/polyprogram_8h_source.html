112  template <class Iterator>
113  PolyOperation(const PolyOperationDesc &desc, Iterator start, Iterator end, std::shared_ptr<pisa::poly::PolyProgram> parent)
114  {
115  m_description = desc.desc;
116  operation_name = desc.name;
117  setParentProgram(parent);
118  setOperationName(desc.name);
119 
120  std::tuple<std::string, int, int> decomposed;
121  int index = 0;
122  for (auto iter = start; iter != end; ++iter)
123  {
124  PARAM_TYPE operation = m_description.params[index];
125  std::string value = *iter;
126  switch (operation)
127  {
128  case OP_NAME:
129  if (!desc.force_desc_op_name)
130  setOperationName(value);
131  break;
132  case INPUT_ARGUMENT:
133  decomposed = Polynomial::decomposePolyStringForm(value);
134  addInput(std::get<0>(decomposed), std::get<1>(decomposed), std::get<2>(decomposed));
135  break;
136  case OUTPUT_ARGUMENT:
137  decomposed = Polynomial::decomposePolyStringForm(value);
138  addOutput(std::get<0>(decomposed), std::get<1>(decomposed), std::get<2>(decomposed));
139  break;
140  case INPUT_OUTPUT_ARGUMENT:
141  decomposed = Polynomial::decomposePolyStringForm(value);
142  addInput(std::get<0>(decomposed), std::get<1>(decomposed), std::get<2>(decomposed));
143  addOutput(std::get<0>(decomposed), std::get<1>(decomposed), std::get<2>(decomposed));
144  break;
145  case POLYMOD_DEG_LOG2:
146  m_parent_program->setPolyModulusDegree(stoi(value));
147  break;
148  case CIPHER_DEGREE:
149  setCipherDegree(stoi(value));
150  break;
151  case RNS_TERM:
152  setRnsTerms(stoi(value));
153  break;
154  case FHE_SCHEME:
155  m_parent_program->setScheme(fromString(value));
156  break;
157  case PARAM:
158  throw std::runtime_error("Params not current supported by poly_program initializer list");
159  break;
160  case GALOIS_ELT:
161  setGaloisElt(stoi(value));
162  break;
163  case FACTOR:
164  setFactor(stoi(value));
165  break;
166  case KEY_RNS:
167  m_parent_program->setKeyRns(stoi(value));
168  break;
169  case ALPHA:
170  m_parent_program->setAlpha(stoi(value));
171  break;
172  case QSIZE:
173  m_parent_program->setQSize(stoi(value));
174  break;
175  case DNUM:
176  m_parent_program->setDNum(stoi(value));
177  break;
178  }
179  index++;
180  }
181  }
