210 template <typename T>
211 void PISAFunctionalModel<T>::decode(pisa::PISAInstruction *instr)
212 {
213  try
214  {
215 
216  if (this->m_trace_execution)
217  {
218  pisa::PISAInstructionTrace<T> instruction_trace;
219  instruction_trace.setInstruction(instr);
220  m_execution_trace.push_back(instruction_trace);
221 
222  //Grab inputs
223  std::vector<pisa::MultiRegister<T>> inputs;
224  for (int x = 0; x < m_execution_trace.back().instruction()->numInputOperands(); x++)
225  {
226  inputs.push_back(m_memory.copy(instr->getInputOperand(x).location()));
227  }
228  m_execution_trace.back().setExecutionInputs(inputs);
229  }
230 
231  if (instr->Name() == pisa::instruction::Add::baseName)
232  {
233  addInstrDecodeExecute(static_cast<pisa::instruction::Add *>(instr));
234  }
235  else if (instr->Name() == pisa::instruction::Sub::baseName)
236  {
237  subInstrDecodeExecute(static_cast<pisa::instruction::Sub *>(instr));
238  }
239  else if (instr->Name() == pisa::instruction::Mul::baseName)
240  {
241  mulInstrDecodeExecute(static_cast<pisa::instruction::Mul *>(instr));
242  }
243  else if (instr->Name() == pisa::instruction::Muli::baseName)
244  {
245  muliInstrDecodeExecute(static_cast<pisa::instruction::Muli *>(instr));
246  }
247  else if (instr->Name() == pisa::instruction::Mac::baseName)
248  {
249  macInstrDecodeExecute(static_cast<pisa::instruction::Mac *>(instr));
250  }
251  else if (instr->Name() == pisa::instruction::Maci::baseName)
252  {
253  maciInstrDecodeExecute(static_cast<pisa::instruction::Maci *>(instr));
254  }
255  else if (instr->Name() == pisa::instruction::Intt::baseName)
256  {
257  iNttInstrDecodeExecute(static_cast<pisa::instruction::Intt *>(instr));
258  }
259  else if (instr->Name() == pisa::instruction::Ntt::baseName)
260  {
261  nttInstrDecodeExecute(static_cast<pisa::instruction::Ntt *>(instr));
262  }
263  else if (instr->Name() == pisa::instruction::Copy::baseName)
264  {
265  copyInstrDecodeExecute(static_cast<pisa::instruction::Copy *>(instr));
266  }
267  else
268  {
269  throw std::runtime_error("Failed to decode " + instr->Name());
270  }
271 
272  if (this->m_trace_execution)
273  {
274  //Grab results
275  std::vector<pisa::MultiRegister<T>> results;
276  for (int x = 0; x < m_execution_trace.back().instruction()->numOutputOperands(); x++)
277  {
278  results.push_back(m_memory.copy(instr->getOutputOperand(x).location()));
279  }
280  m_execution_trace.back().setExecutionResult(results);
281  }
282  }
283  catch (std::runtime_error err)
284  {
285  std::cout << "Failed to decode instr: " << err.what() << std::endl;
286  throw;
287  }
288 }

290 template <typename T>
291 T PISAFunctionalModel<T>::montgomeryMul(T a, T b, T modulus, bool use_mont)
292 {
293  if (use_mont)
294  {
295  uint64_t u = static_cast<uint64_t>(a) * static_cast<uint64_t>(b);
296  uint64_t k = modulus - 2;
297  uint64_t t = u & std::numeric_limits<uint>::max();
298  uint64_t m = (t * k) & std::numeric_limits<uint>::max();
299  u += m * modulus;
300  u >>= 32;
301  u = u - (u >= modulus) * modulus;
302 
303  return u;
304  }
305  else
306  {
307  return a * b % modulus;
308  }
309 }

311 template <typename T>
312 T PISAFunctionalModel<T>::montgomeryAdd(T a, T b, T modulus, bool use_mont)
313 {
314  if (use_mont)
315  {
316  uint64_t u = a + b;
317 #ifdef DEBUG
318  if (u >= 2 * modulus)
319  throw std::runtime_error("Value out of bounds");
320 #endif
321  if (u < modulus)
322  {
323  u = u;
324  }
325  else
326  {
327  u = u - modulus;
328  }
329  return u;
330  }
331  else
332  {
333  return a + b % modulus;
334  }
335 }

487 template <typename T>
488 void PISAFunctionalModel<T>::iNttInstrDecodeExecute(pisa::instruction::Intt *instr)
489 {
490  try
491  {
492  //Decode
493  pisa::MultiRegister<T> &dst_0 = m_memory[instr->getOutputOperand(0).location()];
494  pisa::MultiRegister<T> &dst_1 = m_memory[instr->getOutputOperand(1).location()];
495  pisa::MultiRegister<T> &src_1 = m_memory[instr->getInputOperand(0).location()];
496  pisa::MultiRegister<T> &src_2 = m_memory[instr->getInputOperand(1).location()];
497 
498  const int lN = instr->PMD();
499  const int N = pow(2, lN);
500  const int half_N = N >> 1;
501  const int block_size = src_1.size() * 2;
502  const int half_block = src_1.size();
503  const int starting_offset = block_size * instr->wParam().block();
504  const auto modulus = m_modulus_chain[instr->wParam().residual()];
505  const auto stage = instr->wParam().stage();
506  // TODO: discuss how to handle twiddle map key values - currently just "{ge}", it could only be the ge value instead..
507  const std::string ge = std::to_string(instr->galois_element());
508  const int block_count = lN - 14;
509  const int increment = pow(2, block_count);
510 
511  const int slice_size = half_N / increment;
512  const int start = instr->wParam().block() * slice_size;
513  const int end = start + slice_size;
514 
515  if (m_bit_reverse_table.size() == 0)
516  {
517  generate_bit_reverse_table(N, lN);
518  }
519 
520 #pragma omp parallel for
521  for (uint32_t i = start; i < end; i += 1)
522  {
523  auto j = m_bit_reverse_table[i]; //
524 
525  int in0 = i % half_block;
526  int in1 = ((i + half_N) % half_block) + half_block;
527  int out0 = 2 * i % block_size;
528  int out1 = (2 * i + 1) % block_size; //Reads from input
529 
530  int sp = lN - 1 - stage;
531  int k = (j >> sp) << sp;
532 
533  //Resolve input and output locations based on index and spread across two inputs)
534  T Xin_0 = in0 < half_block ? (src_1.data()[in0]) : (src_2.data()[in0 - half_block]);
535  T Xin_1 = in1 < half_block ? (src_1.data()[in1]) : (src_2.data()[in1 - half_block]);
536  T &Xout_0 = out0 < half_block ? (dst_0.data()[out0]) : (dst_1.data()[out0 - half_block]);
537  T &Xout_1 = out1 < half_block ? (dst_0.data()[out1]) : (dst_1.data()[out1 - half_block]);
538 
539  T t0 = Xin_0;
540  T t1 = montgomeryMul(Xin_1, m_twiddle_intt.at(ge)[instr->wParam().residual()][k], modulus, true); // Need montgomery mul single op
541  T t2 = modulus - t1;
542 
543  Xout_0 = montgomeryAdd(t0, t1, modulus, true);
544  Xout_1 = montgomeryAdd(t0, t2, modulus, true);
545  }
546 
547  return;
548  }
549  catch (std::runtime_error err)
550  {
551  std::cout << "INtt runtime error:" << err.what() << std::endl;
552  throw err;
553  }
554  catch (...)
555  {
556  throw;
557  }
558 }

560 template <typename T>
561 void PISAFunctionalModel<T>::nttInstrDecodeExecute(pisa::instruction::Ntt *instr)
562 {
563  try
564  {
565  //Decode
566  pisa::MultiRegister<T> &dst_0 = m_memory[instr->getOutputOperand(0).location()];
567  pisa::MultiRegister<T> &dst_1 = m_memory[instr->getOutputOperand(1).location()];
568  pisa::MultiRegister<T> &src_1 = m_memory[instr->getInputOperand(0).location()];
569  pisa::MultiRegister<T> &src_2 = m_memory[instr->getInputOperand(1).location()];
570 
571  const int lN = instr->PMD();
572  const int N = pow(2, lN);
573  const int half_N = N >> 1;
574  const int block_size = src_1.size() * 2;
575  const int half_block = src_1.size();
576  const int starting_offset = block_size * instr->wParam().block();
577 
578  const int block_count = lN - 14;
579  const int increment = pow(2, block_count);
580 
581  if (m_start_table.size() == 0)
582  {
583  createStartTable(increment);
584  }
585 
586  const int start = m_start_table[instr->wParam().block()];
587  const auto residual = instr->wParam().residual();
588  const auto modulus = m_modulus_chain[residual];
589  const auto stage = instr->wParam().stage();
590 
591  //Compute the bit reversal table if needed
592  if (m_bit_reverse_table.size() == 0)
593  {
594  generate_bit_reverse_table(N, lN);
595  }
596 
597 #pragma omp parallel for
598  for (uint32_t i = start; i < half_N; i += increment)
599  {
600  auto j = m_bit_reverse_table[i]; // Look up precomputed bit_reverse values
601 
602  int in0 = (2 * j) % block_size;
603  int in1 = (2 * j + 1) % block_size;
604  int out0 = j % half_block;
605  int out1 = ((j + half_N) % half_block) + half_block;
606 
607  int sp = lN - 1 - stage;
608  int k = (j >> sp) << sp;
609 
610  //Resolve input and output locations based on index and spread across two inputs)
611  T Xin_0 = in0 < half_block ? (src_1.data()[in0]) : (src_2.data()[in0 - half_block]);
612  T Xin_1 = in1 < half_block ? (src_1.data()[in1]) : (src_2.data()[in1 - half_block]);
613  T &Xout_0 = out0 < half_block ? (dst_0.data()[out0]) : (dst_1.data()[out0 - half_block]);
614  T &Xout_1 = out1 < half_block ? (dst_0.data()[out1]) : (dst_1.data()[out1 - half_block]);
615 
616  T t0 = Xin_0;
617  T t1 = 0;
618  if (stage == 0)
619  {
620  t1 = Xin_1;
621  }
622  else
623  {
624  t1 = montgomeryMul(Xin_1, m_twiddle_ntt[residual][k], modulus, true); // Need montgomery mul single op
625  }
626  T t2 = modulus - t1;
627 
628  Xout_0 = montgomeryAdd(t0, t1, modulus, true);
629  Xout_1 = montgomeryAdd(t0, t2, modulus, true);
630  }
631 
632  return;
633  }
634  catch (std::runtime_error err)
635  {
636  std::cout << "Ntt runtime error:" << err.what() << std::endl;
637  throw err;
638  }
639  catch (...)
640  {
641  throw;
642  }
643 }

819 {

821  {
822  std::string current_line;
823  while (std::getline(input_stream, current_line))
824  {
825  std::vector<std::string> components;
826  std::istringstream current_line_ss(current_line);
827 
828  std::string component;
829  while (std::getline(current_line_ss, component, ','))
830  {
831  if (component != "\r" && component != "")
832  components.push_back(std::move(component));
833  }
834  if (components[0] == "memory")
835  {
836  std::vector<T> values;
837  std::transform((components.begin() + 2), components.end(), std::back_inserter(values), [this](auto s) {
838  return std::stoi(s);
839  });
840  m_memory[components[1]].setData(values);
841  }
842  if (components[0] == "modulus_chain")
843  {
844  std::vector<T> values;
845  std::transform((components.begin() + 1), components.end(), std::back_inserter(values), [this](auto s) {
846  return std::stoi(s);
847  });
848  this->setModulusChain(values);
849  }
850  if (components[0] == "ntt")
851  {
852  if (m_twiddle_ntt.size() < std::stoi(components[1]))
853  {
854  m_twiddle_ntt.resize(std::stoi(components[1]));
855  }
856 
857  std::vector<T> values;
858  std::transform((components.begin() + 2), components.end(), std::back_inserter(values), [this](auto s) {
859  return std::stoi(s);
860  });
861 
862  m_twiddle_ntt[std::stoi(components[1])] = values;
863  }
864  if (components[0] == "intt")
865  {
866  if (m_twiddle_intt.count(components[1]) == 0)
867  {
868  m_twiddle_intt[components[1]];
869  }
870  auto intt_values = m_twiddle_intt[components[1]];
871  if (intt_values.size() < std::stoi(components[2]))
872  {
873  intt_values.resize(std::stoi(components[2]));
874  }
875 
876  std::vector<T> values;

878  return std::stoi(s);

882  }
