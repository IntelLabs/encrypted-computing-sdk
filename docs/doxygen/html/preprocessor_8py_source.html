10 def __dependencyGraphForVars(insts_list: list) -> (nx.Graph, set, set):
11  """
12  Given the listing of instructions, this method returns the dependency graph
13  for the variables in the listing and the sets of destination and source variables.
14 
15  Parameters:
16  insts_list (list): List of corresponding pre-processed P-ISA instructions containing the variables
17  to process.
18 
19  Returns:
20  tuple: A tuple containing:
21  - nx.Graph: Dependency graph for the variables in the listing.
22  Nodes: variable names.
23  Edges: dependencies among variables.
24  Dependencies:
25  - All variables being read from at the same time in any one instruction
26  depend on each other because it is forbidden to read more than once from
27  the same register bank in the same instruction.
28  - All variables being written to at the same time in any one instruction
29  depend on each other because it is forbidden to write more than once to
30  the same register bank in one instruction.
31  - set: Set of all variables (name) that are destinations in the input `insts_list`.
32  - set: Set of all variables (name) that are sources in the input `insts_list`.
33  """
34  retval = nx.Graph()
35  all_dests_vars = set()
36  all_sources_vars = set()
37 
38  for inst in insts_list:
39  extra_sources = []
40  for idx, v in enumerate(inst.dests):
41  all_dests_vars.add(v.name)
42  if v.name not in retval:
43  retval.add_node(v.name)
44  for v_i in range(idx + 1, len(inst.dests)):
45  v_next = inst.dests[v_i]
46  if v.name == v_next.name:
47  raise RuntimeError(f"Cannot write to the same variable in the same instruction more than once: {inst.toPISAFormat()}")
48  if not retval.has_edge(v.name, v_next.name):
49  retval.add_edge(v.name, v_next.name)
50  # Mac deps already handled in the Mac instructions themselves
51  # if isinstance(inst, (xinst.Mac, xinst.Maci)):
52  # extra_sources.append(v)
53 
54  inst_all_sources = extra_sources + inst.sources
55  for idx, v in enumerate(inst_all_sources):
56  all_sources_vars.add(v.name)
57  if v.name not in retval:
58  retval.add_node(v.name)
59  for v_i in range(idx + 1, len(inst_all_sources)):
60  v_next = inst_all_sources[v_i]
61  if v.name == v_next.name:
62  raise RuntimeError(f"Cannot read from the same variable in the same instruction more than once: {inst.toPISAFormat()}")
63  if not retval.has_edge(v.name, v_next.name):
64  retval.add_edge(v.name, v_next.name)
65 
66  return retval, all_dests_vars, all_sources_vars

68 def injectVariableCopy(mem_model: MemoryModel,
69  insts_list: list,
70  instruction_idx: int,
71  var_name: str) -> int:
72  """
73  Injects a copy of a variable into the instruction list at the specified index.
74 
75  Parameters:
76  mem_model (MemoryModel): The memory model containing the variables.
77  insts_list (list): The list of instructions.
78  instruction_idx (int): The index at which to inject the copy.
79  var_name (str): The name of the variable to copy.
80 
81  Returns:
82  int: Index for the instruction in the list after injection.
83 
84  Raises:
85  IndexError: If the instruction index is out of range.
86  """
87  if instruction_idx < 0 or instruction_idx >= len(insts_list):
88  raise IndexError(f'instruction_idx: Expected index in range [0, {len(insts_list)}), but received {instruction_idx}.')
89  last_instruction: XInstruction = insts_list[instruction_idx]
90  last_instruction_sources = last_instruction.sources[:]
91  for idx, variable in enumerate(last_instruction_sources):
92  if variable.name == var_name:
93  # Find next available temp var name
94  temp_name = mem_model.findUniqueVarName()
95  temp_var = mem_model.retrieveVarAdd(temp_name, -1)
96  # Copy source var into temp
97  copy_xinst = xinst.Copy(id = last_instruction.id[1],
98  N = 0,
99  dst = [ temp_var ],
100  src = [ variable ],
101  comment='Injected copy for bank reduction.')
102  insts_list.insert(instruction_idx, copy_xinst)
103  # Replace src by temp
104  last_instruction.sources[idx] = temp_var
105  instruction_idx += 1
106 
107  return instruction_idx

109 def reduceVarDepsByVar(mem_model: MemoryModel,
110  insts_list: list,
111  var_name: str):
112  """
113  Reduces variable dependencies by injecting copies of the specified variable.
114 
115  Parameters:
116  mem_model (MemoryModel): The memory model containing the variables.
117  insts_list (list): The list of instructions.
118  var_name (str): The name of the variable to reduce dependencies for.
119  """
120  last_pos = 0
121  last_instruction = None
122  # Find all instructions* with specified variable and make it a copy
123  # * care with mac instructions
124  while last_pos < len(insts_list):
125  if var_name in (v.name for v in insts_list[last_pos].sources):
126  last_instruction: XInstruction = insts_list[last_pos]
127  if isinstance(last_instruction, (xinst.Mac, xinst.Maci)):
128  # Check if the conflicting variable is the accumulator
129  if last_instruction.sources[0].name == var_name:
130  # Turn all other variables into copies
131  for variable in last_instruction.sources[1:]:
132  last_pos = injectVariableCopy(mem_model, insts_list, last_pos, variable.name)
133  assert last_instruction == insts_list[last_pos]
134  last_instruction = None # avoid further processing of instruction
135  last_pos += 1
136  continue
137  # If conflict variable was not the accumulator, proceed to change the other variables
138  # Skip copy, twxntt and xrshuffle
139  if not isinstance(last_instruction, (xinst.twiNTT,
140  xinst.twiNTT,
141  xinst.irShuffle,
142  xinst.rShuffle,
143  xinst.Copy)):
144  # Break up indicated variable in sources into a temp copy
145  last_pos = injectVariableCopy(mem_model, insts_list, last_pos, var_name)
146  assert last_instruction == insts_list[last_pos]
147 
148  last_pos += 1

150 def assignRegisterBanksToVars(mem_model: MemoryModel,
151  insts_list: list,
152  use_bank0: bool,
153  verbose = False) -> str:
154  """
155  Assigns register banks to variables using vertex coloring graph algorithm.
156 
157  The variables contained in the MemoryModel object will be modified to reflect
158  their suggested bank.
159 
160  Parameters:
161  mem_model (MemoryModel): The MemoryModel object, where all variables are kept. Variables detected that are
162  not already in the MemoryModel collection of variables will be added automatically.
163  The variables contained in the MemoryModel object will be modified to reflect
164  their suggested bank.
165  insts_list (list): List of corresponding pre-processed P-ISA instructions containing the variables
166  to process.
167  use_bank0 (bool): All variables are written into registers in bank 0 from SPAD, while no XInst
168  should be writing its results in bank 0 to avoid write-write conflicts.
169  If `True`, variables that can remain in bank 0 will be kept there (variables
170  that are never written to).
171  If `False`, bank 0 will not be assigned to any variable. Resulting ASM instructions
172  should add corresponding `move` instructions to move variables from bank 0 to
173  correct bank.
174  verbose (bool, optional): If True, prints verbose output. Defaults to False.
175 
176  Raises:
177  ValueError: Thrown for these cases:
178  - Invalid input values for parameters.
179  - Variables in listing cannot be successfully assigned to banks (the number
180  of banks is insufficient to accommodate the listing given the rules).
181  This should not happen, as long as rules are respected, and all instructions
182  have, at most, 3 inputs, and at most, 3 outputs.
183 
184  Returns:
185  str: The unique dummy variable name that was not used by the collection of variables in the
186  instruction listing.
187  """
188  reduced_vars = set()
189  needs_reduction = True
190  pass_counter = 0
191  while needs_reduction:
192  pass_counter += 1
193  if verbose:
194  print(f"Pass {pass_counter}")
195  # Extract the dependency graph for variables
196  dep_graph_vars, dest_names, source_names = __dependencyGraphForVars(insts_list)
197  only_sources = source_names - dest_names # Find which variables are ever only used as sources
198  color_dict = nx.greedy_color(dep_graph_vars) # Do coloring
199 
200  needs_reduction = False
201  for var_name, bank in color_dict.items():
202  if bank > 2:
203  if var_name in reduced_vars:
204  raise RuntimeError(('Found invalid bank {} > 2 for variable {} already reduced.').format(bank,
205  var_name))
206  # DEBUG print
207  if verbose:
208  print('Variable {} ({}) requires reduction.'.format(var_name, bank))
209  reduceVarDepsByVar(mem_model, insts_list, var_name)
210  reduced_vars.add(var_name) # Track reduced variable
211  needs_reduction = True
212 
213  # Assign banks based on coloring algo results
214  for v in mem_model.variables.values():
215  if not mem_model.isMetaVar(v.name): # Skip meta variables
216  assert(v.name in color_dict)
217  bank = color_dict[v.name]
218  assert bank < 3, f'{v.name}, {bank}'
219  # If requested, keep vars used only as sources in bank 0
220  v.suggested_bank = bank + (0 if use_bank0 and (v.name in only_sources) else 1)
221 
222  retval: str = mem_model.findUniqueVarName()
223 
224  return retval
