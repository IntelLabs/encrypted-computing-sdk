140  for (int x = 1; x < op_tokens.size() - 1; x++)
141  {
142  instruction_trace.push_back(createInstr(op_tokens[x], intermediate_regs[next_reg], intermediate_regs[output_reg], intermediate_regs[rand() % intermediate_regs.size()]));
143  output_reg = next_reg;
144  next_reg = rand_r(&random_seed) % intermediate_regs.size();
145  }
146 
147  //create final instruction

149  if (op_tokens.size() > 1)
150  {
151  instruction_trace.push_back(createInstr(final_token, output_regs[0], intermediate_regs[output_reg], intermediate_regs[rand() % intermediate_regs.size()]));
152  }
153 
154  // }
