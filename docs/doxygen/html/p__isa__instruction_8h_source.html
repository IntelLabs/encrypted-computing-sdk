278  break;
279  case pisa::INPUT_OPERAND:
280  stream << instr.getInputOperand(input_count);
281  input_count++;
282  break;
283  case pisa::OUTPUT_OPERAND:
284  stream << instr.getOutputOperand(output_count);
285  output_count++;
286  break;
287  case pisa::INPUT_OUTPUT_OPERAND:
288  stream << instr.getOutputOperand(output_count);
289  input_count++;
290  output_count++;

294  break;
295  case pisa::RESIDUAL:

297  break;
298  case pisa::W_PACKED_PARAM:

301  case pisa::IMMEDIATE:

303  input_count++;
