28 // Helper function for generating the kernel generator input given an operation
29 inline std::string genKernInput(const pisa::poly::PolyOperation &op)
30 {
31  /*
32  * CONTEXT SCHEME poly_order key_rns current_rns
33  * DATA symbol num_parts
34  * OPNAME output input(s)
35  */
36  std::ostringstream input;
37  // CONTEXT
38  input << "CONTEXT " << toString(op.parentProgram()->scheme())
39  << " " << op.parentProgram()->getPolyModulusDegree()
40  << " " << op.parentProgram()->getKeyRns()
41  << " " << op.getInputOperand(0).num_of_rns_terms
42  << "\n";
43 
44  // DATA
45  // NOTE: CipherDegree is tied to HEOperation not Operand
46  auto operation = op.Name();
47  std::vector<pisa::poly::Polynomial> inputs = op.getInputLocations();
48  std::vector<pisa::poly::Polynomial> output = op.getOutputLocations();
49 
50  //#TODO Naming is switched to use generic sequential naming allowing program mapper to control the final naming
51  //This allows generic kernels to be generated effectively for the cache, while passing names worked well for single
52  //operations it creates issues for multi ops. A more robust fix with runtime control on which naming control scheme to use may be desirable.
53  for (int i = 0; i < op.numOutputOperands(); ++i)
54  {
55  input << "DATA " << /*output[i].register_name*/ "output" << i
56  << " " << output[i].num_of_polynomials << "\n";
57  }
58 
59  for (int i = 0; i < op.numInputOperands(); ++i)
60  {
61  input << "DATA " << /*inputs[i].register_name*/ "input" << i
62  << " " << inputs[i].num_of_polynomials << "\n";
63  }
64 
65  // OP
66  input << std::uppercase << op.Name() << std::nouppercase;
67  for (int i = 0; i < op.numOutputOperands(); ++i)
68  {
69  input << " "
70  << "output" << i /*output[i].register_name*/;
71  }
72  for (int i = 0; i < op.numInputOperands(); ++i)
73  {
74  input << " "
75  << "input" << i /*inputs[i].register_name*/;
76  }
77 
78  return input.str();
79 }

462 void pisa::kernel::PISAKernel::determineVariableNamingViaGraph()
463 {
464  auto instruction_graph = graph::Graph<pisa::PISAInstruction>::createGraph(instructions);
465  auto inputs = instruction_graph.getInputNodes(true, false, false);
466  auto outputs = instruction_graph.getOutputNodes();
467  auto immediates = instruction_graph.getInputNodes(false, true, false);
468 
469  auto non_repeat_inputs = nonRepeatingRootsNode(inputs);
470  auto non_repeat_outputs = nonRepeatingRootsNode(outputs);
471  for (auto &input : non_repeat_inputs)
472  {
473  input_names.push_back(input);
474  naming_map[input] = input;
475  }
476  // temporary solution for two corner cases:
477  // Two inputs are needed but "d" is parsed before "c" causing input1 and input2 being swapped
478  // Trace has TWO inputs but only "d" is used, it will cause mapInput function to crash
479  // NOTE: This could still lead to an issue for second corner case where only "c" is used but trace has two inputs
480  //if (input_names.size() == 1 && input_names.front() == "d")
481  // input_names.push_back("c");
482  // std::sort(input_names.begin(), input_names.end(), comp);
483  // Attempt #2, just sort inputs if the label contains "input"
484  // Leave the rest of the list alone
485  // TODO: Generalize Function
486  auto comp = [](const std::string &a, const std::string &b) {
487  if (containsString(a, "input") && containsString(b, "input"))
488  return a < b;
489  else
490  return false;
491  };
492  std::sort(input_names.begin(), input_names.end(), comp);
493 
494  for (auto &output : non_repeat_outputs)
495  {
496  output_names.push_back(output);
497  naming_map[output] = output;
498  }
499  std::sort(output_names.begin(), output_names.end());
500 
501  for (auto &immediate : immediates)
502  {
503  if (immediate_map.count(immediate.GetDat().label) == 0)
504  {
505  immediate_names.push_back(immediate.GetDat().label);
506  }
507  immediate_map[immediate.GetDat().label] = immediate.GetDat().label;
508  }
