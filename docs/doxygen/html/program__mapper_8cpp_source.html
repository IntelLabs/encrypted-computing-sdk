8 template <typename T>
9 void ProgramMapper<T>::generatePisaProgramFromHEProgram(std::shared_ptr<pisa::poly::PolyProgram> program_trace)
10 {
11  // Generates a map from raw seal trace to V0 trace to allow for
12  // input/output variable name alignment with trace file to support
13  // validation.
14  std::map<std::string, std::string> register_map;
15 
16  auto program_graph = graph::Graph<pisa::poly::PolyOperation>::createGraph(program_trace->operationsRaw());
17  auto program_inputs = program_graph.getInputNodes(true, true, false);
18  auto program_outputs = program_graph.getOutputNodes();
19  for (auto input : program_inputs)
20  {
21  register_map[input.GetDat().label] = input.GetDat().label;
22  }
23  for (auto output : program_outputs)
24  {
25  register_map[output.GetDat().label] = output.GetDat().label;
26  }
27 
28  if (arguments.enable_intermediates == true)
29  {
30  auto all_program_data_nodes = program_graph.getDataGraph();
31  for (auto data : all_program_data_nodes.getNodes())
32  {
33  register_map[data.GetDat().label] = data.GetDat().label;
34  }
35  }
36 
37  // Generates map mapping all input and output locations in the program
38  // trace not part of the program_trace input/output set to include
39  // operation namespace.
40  // This is used for efficiently linking HE_Operations together.
41  // Currently input/output operations are determined via the trace @TODO
42  // better to switch this to graph based approach
43  // and simply raise warnings when there is a mismatch between graph and
44  // trace. A middle term solution maybe to create option to use a handinput_parser_v0
45  // aligned map.
46  register_map = mapProgramTraceOperationsIntoRegisterMap(program_trace->operations(), register_map);
47 
48  // Generate PISAKernels(ninja kernels) as needed by the program trace.
49  // Checks for each kernel the HE operation and parameters and calls
50  // kerngen.py to generate the ninja kernel if it does not already
51  // exist in the kernel_cache.
52 
53  int max_rns_term = 0;
54  auto p_isa_kernels = generatePISAKernelsFromHEOperationVector(program_trace->operations(), arguments.kerngen, max_rns_term);
55 
56  /* Remaps kernel input/output names based upon the trace mapping There
57  * exists a 1 : 1 mapping between HEOperation -> PISAKernel, this
58  * function aligns the PISA input/output names to match HEOperation
59  */
60  mapKernelInputOutputToRegisterMap(p_isa_kernels, program_trace->operations(), register_map);
61 
62  // Generates a vector containing all of the remapped instructions for
63  // all of the p_isa_kernels resulting in a single set of all P-ISA
64  // instructions necessary to execute
65  // the operations specified in the seal trace.
66  auto combined_instructions = outputCombinedPisaInstructions(p_isa_kernels, arguments.apply_name_spacing);
67 
68  /* Apply instruction graph rewritter and instruction hardware fixes and optimization */
69  PISAGraphOptimizer graph_optimizer;
70  graph_optimizer.applyDuplicateInputVariableSeparation(combined_instructions);
71 
72  /* generates graph from combined p_isa instructions */
73  auto p_isa_graph = graph::Graph<pisa::PISAInstruction>::createGraph(combined_instructions);
74  combined_instructions = graph_optimizer.generateInstructionStreamFromGraph(p_isa_graph, true, combined_instructions);
75 
76  /* Re-generate combined instructions as needed */
77  if (arguments.generated_name != "")
78  {
79  PisaTestGenerator test_gen;
80  auto generated_json = test_gen.generateJSONForGraph(p_isa_graph);
81  test_gen.populateCalculatedOutputResults(combined_instructions, generated_json);
82  test_gen.writeJSON(generated_json, arguments.generated_name);
83  }
84 
85  /* renders instructions from graph, Current renders two graphs, the
86  * seal program trace graph at the HEOperation level and the P-ISA
87  * instruction level graph*/
88  if (arguments.generate_graphs)
89  {
90 
91  if (arguments.export_dot)
92  {
93  std::cout << "Writing graph to dot file: " << arguments.dot_file_name << std::endl;
94  std::string dot_file_name = arguments.outfile_prefix.replace_extension("dot");
95 #pragma omp parallel sections
96  {
97 #pragma omp section
98  program_graph.writeDotFile(arguments.dot_file_name, graph::NAME);
99 #pragma omp section
100  p_isa_graph.writeDotFile(dot_file_name, graph::NAME);
101  } // End of parallel region
102 
103  auto inputs = p_isa_graph.getInputNodes();
104  std::cout << "P_ISA Graph Input Nodes\n"
105  << graph::with_delimiter(inputs, "\n");
106  }
107  }
108 
109  /* Outputs the combined p_isa_kernel to an instruction stream,
110  * PISAInstructions are implemented so that they print out functionally
111  * with stream operator
112  */
113  const auto final_kernel_filename = arguments.outfile_prefix.replace_extension("csv");
114  writeToFileBy(final_kernel_filename.string(), combined_instructions,
115  [membank = arguments.output_memory_bank](auto *instruction) {
116  instruction->setOutputBlock(membank);
117  return *instruction;
118  });
119 
120  if (arguments.verbose)
121  {
122  for (const auto &instruction : combined_instructions)
123  std::cout << *instruction << std::endl;
124  }
125 
126  /* generates memory file for p_isa_graph */
127  std::vector<std::string> mem_file = generateMemFile(p_isa_graph, max_rns_term);
128  const auto memory_filename = arguments.outfile_prefix.replace_extension("tw.mem");
129  writeToFileBy(memory_filename.string(), mem_file, [](const auto &line) { return line; });
130 
131  return;
132 }

197 std::map<std::string, std::string> ProgramMapper<T>::mapProgramTraceOperationsIntoRegisterMap(std::vector<std::shared_ptr<pisa::poly::PolyOperation>> program_trace,
198  std::map<std::string, std::string> register_map)
199 {
200 
201  for (int x = 0; x < program_trace.size(); x++)
202  {
203  std::string location;
204  for (int y = 0; y < program_trace[x]->numInputOperands(); y++)
205  {
206  location = program_trace[x]->getInputOperand(y).location();
207  std::string value = register_map[location];
208  if (value.size() == 0)
209  {
210  register_map[location] = program_trace[x]->Name() + std::to_string(x) + "input" + std::to_string(y);
211  }
212  }
213  for (int y = 0; y < program_trace[x]->numOutputOperands(); y++)
214  {
215  location = program_trace[x]->getOutputOperand(y).location();
216  std::string value = register_map[location];
217  if (value.size() == 0)
218  {
219  register_map[location] = program_trace[x]->Name() + std::to_string(x) + "output" + std::to_string(y);
220  }
221  }
222  }
223  return register_map;
