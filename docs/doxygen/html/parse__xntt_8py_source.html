12  op_name: str,
13  tw_separator: str) -> Namespace:
14  """
15  Parses an `xntt` instruction from a P-ISA kernel instruction string.
16 
17  Parameters:
18  line (str): The line containing the instruction to parse.
19  Instruction format: N, op_name, dst0, dst1, src0, src1, twiddle, res # comment
20  Comment is optional.
21 
22  op_name (str): The operation name that should be contained in the line.
23 
24  tw_separator (str): The separator used in the twiddle information.
25 
26  Returns:
27  Namespace: A namespace with the following attributes:
28  N (int): Ring size = Log_2(PMD)
29  op_name (str): Operation name
30  dst (list of tuple): List of destinations of the form (variable_name, suggested_bank).
31  src (list of tuple): List of sources of the form (variable_name, suggested_bank).
32  res (int): Residual for the operation.
33  stage (int): Stage number of the current NTT instruction.
34  block (int): Index of current word in the 2-words (16KB) polynomial.
35  comment (str): String with the comment attached to the line (empty string if no comment).
36 
37  None: If an `xntt` could not be parsed from the input.
38  """
39 
40  OP_NUM_DESTS = 2
41  OP_NUM_SOURCES = 2
42  OP_NUM_TOKENS = 8
43 
44  retval = None
45  tokens = xinst.XInstruction.tokenizeFromPISALine(op_name, line)
46  if tokens:
47  retval = {"comment": tokens[1]}
48  instr_tokens = tokens[0]
49 
50  if len(instr_tokens) > OP_NUM_TOKENS:
51  warnings.warn(f'Extra tokens detected for instruction "{op_name}"', SyntaxWarning)
52 
53  retval["N"] = int(instr_tokens[0])
54  retval["op_name"] = instr_tokens[1]
55  params_start = 2
56  params_end = params_start + OP_NUM_DESTS + OP_NUM_SOURCES
57  dst_src = xinst.XInstruction.parsePISASourceDestsFromTokens(instr_tokens,
58  OP_NUM_DESTS,
59  OP_NUM_SOURCES,
60  params_start)
61  retval.update(dst_src)
62  twiddle = instr_tokens[params_end]
63  retval["res"] = int(instr_tokens[params_end + 1])
64 
65  # Parse twiddle (w_<res>_<stage>_<block>, where "_" is the `tw_separator`)
66  twiddle_tokens = list(map(lambda s: s.strip(), twiddle.split(tw_separator)))
67  if len(twiddle_tokens) != 4:
68  raise ValueError(f'Error parsing twiddle information for "{op_name}" in line "{line}".')
69  if twiddle_tokens[0] != "w":
70  raise ValueError(f'Invalid twiddle detected for "{op_name}" in line "{line}".')
71  if int(twiddle_tokens[1]) != retval["res"]:
72  raise ValueError(f'Invalid "residual" component detected in twiddle information for "{op_name}" in line "{line}".')
73  retval["stage"] = int(twiddle_tokens[2])
74  retval["block"] = int(twiddle_tokens[3])
75 
76  retval = Namespace(**retval)
77  assert(retval.op_name == op_name)
78  return retval
