106 void PISAGraphOptimizer::applyDuplicateInputVariableSeparation(std::vector<pisa::PISAInstruction *> &instr_order)
107 {
108  std::vector<pisa::PISAInstruction *> newOrder;
109  for (auto instr : instr_order)
110  {
111  //Check input for matches
112  bool match = false;
113  std::pair<int, int> matching_indices;
114 
115  if (instr->numInputOperands() == 2)
116  {
117  auto &operand_0 = instr->getInputOperand(0);
118  auto &operand_1 = instr->getInputOperand(1);
119  if (operand_0.location() == operand_1.location())
120  {
121  std::cout << "Duplicate input variable detected" << std::endl;
122  match = true;
123  }
124  }
125  else if (instr->numInputOperands() == 3)
126  {
127  auto &operand_0 = instr->getInputOperand(0);
128  auto &operand_1 = instr->getInputOperand(1);
129  auto &operand_2 = instr->getInputOperand(2);
130 
131  if (operand_0.location() == operand_1.location())
132  {
133  match = true;
134  matching_indices = std::pair<int, int>(0, 1);
135  }
136  if (operand_0.location() == operand_2.location())
137  {
138  match = true;
139  matching_indices = std::pair<int, int>(0, 2);
140  }
141  if (operand_1.location() == operand_2.location())
142  {
143  match = true;
144  matching_indices = std::pair<int, int>(1, 2);
145  }
146  }
147  if (match == false)
148  {
149  newOrder.push_back(instr);
150  }
151  else
152  {
153  std::cout << "Duplicate input variable detected" << std::endl;
154  auto copy_instr = pisa::instruction::Copy().create();
155  copy_instr->setPMD(instr->PMD());
156  copy_instr->setResidual(instr->residual());
157  copy_instr->addInputOperand(instr->getInputOperand(matching_indices.second));
158  auto output_operand = instr->getInputOperand(matching_indices.second);
159  output_operand.setLocation("copyA" + output_operand.location());
160  copy_instr->addOutputOperand(output_operand);
161  newOrder.push_back(copy_instr);
162  instr->getInputOperand(matching_indices.second).setLocation(output_operand.location());
163  newOrder.push_back(instr);
164  }
165  }
166  instr_order = newOrder;
167  return;
168 }
