404  def _get_cycle_ready(self):
405  """
406  Returns the current value for ready cycle.
407 
408  This method is called by property cycle_ready getter to retrieve the value.
409  An instruction cycle ready value is the maximum among its own and all the
410  sources ready cycles, and destinations (special case).
411 
412  Cycles are measured as tuples: (bundle: int, clock_cycle: int)
413 
414  Overrides `CycleTracker._get_cycle_ready`.
415 
416  Returns:
417  CycleType: The current value for ready cycle.
418  """
419 
420  # we have to be careful that `max` won't iterate on our CycleType tuples' inner values
421  retval = super()._get_cycle_ready()
422  if self.sourcessourcessourcessources:
423  retval = max(retval, *(src.cycle_ready for src in self.sourcessourcessourcessources))
424  if self.destsdestsdestsdests:
425  # dests cycle ready is a special case:
426  # dests are ready to be read or writen to at their cycle_ready, but instructions can
427  # start the following cycle when their dests are ready minus the latency of
428  # the instruction because the dests will be writen to in the last cycle of
429  # the instruction:
430  # Cycle decode_phase write_phase dests_ready latency
431  # 1 INST1 5
432  # 2 INST2 5
433  # 3 INST3 5
434  # 4 INST4 5
435  # 5 INST6 INST1 5
436  # 6 INST7 INST2 INST1 5
437  # 7 INST8 INST3 INST2 5

440  # following INST1, 2, because INST2 will write to the same dest in cycle 6.
