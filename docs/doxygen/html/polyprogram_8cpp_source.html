198 void PolyOperation::setComponents(const heracles::fhe_trace::Instruction &instr_pb)
199 {
200  for (const auto &src : instr_pb.args().dests())
201  addOutput(src.symbol_name(), src.num_rns(), src.order());
202  for (const auto &src : instr_pb.args().srcs())
203  addInput(src.symbol_name(), src.num_rns(), src.order());
204  setRnsTerms(instr_pb.args().srcs(0).num_rns());
205  setCipherDegree(instr_pb.args().srcs(0).order());
206  for (const auto &[k, v] : instr_pb.args().params())
207  {
208  if (k == "galois_elt")
209  setGaloisElt(stoi(v.value()));
210  if (k == "factor")
211  setFactor(stoi(v.value()));
212  }

214 heracles::fhe_trace::Instruction *PolyOperation::getProtobuffFHETraceInstruction()
215 {
216  heracles::fhe_trace::Instruction *instruction = new heracles::fhe_trace::Instruction();
217 
218  instruction->set_op(this->Name());
219  for (auto &output : this->output_locations)
220  {
221  auto dest = instruction->mutable_args()->add_dests();
222  dest->set_symbol_name(output.register_name);
223  dest->set_order(output.num_of_polynomials);
224  dest->set_num_rns(output.num_of_rns_terms);
225  }
226  for (auto &input : this->input_locations)
227  {
228  auto src = instruction->mutable_args()->add_srcs();
229  src->set_symbol_name(input.register_name);
230  src->set_order(input.num_of_polynomials);
231  src->set_num_rns(input.num_of_rns_terms);
232  }
233 
234  // Parse through description and process any special arguments
235  auto desc = this->description();
236  //TODO: additional arguments not currently supported.
237  for (auto param : desc.params)
238  {
239  if (param == GALOIS_ELT)
240  {
241  *((*instruction->mutable_args()->mutable_params())["galois_elt"].mutable_value()) = std::to_string(this->galois_elt);
242  }
243  if (param == FACTOR)
244  {
245  *((*instruction->mutable_args()->mutable_params())["factor"].mutable_value()) = std::to_string(this->factor);
246  }
247  }
248 
249  return instruction;
250 }
