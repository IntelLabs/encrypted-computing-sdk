80 void PISAProgramRuntime<T>::executeProgram(const std::vector<pisa::PISAInstruction *> &instructions)
81 {
82  try
83  {
84  if (m_debug_mode)
85  {
86  m_model.enableExecutionTrace();
87  }
88 
89  for (const auto &instruction : instructions)
90  {
91  executeInstruction(instruction);
92  }
93 
94  if (m_debug_mode)
95  {
96  m_model.stopExecutionTrace();
97 
98  auto trace = m_model.executionTrace();
99  for (auto instr : trace)
100  {
101  instr.printInstructionTrace();
102  }
103  }
104  }
105  catch (std::runtime_error err)
106  {
107  if (m_debug_mode)
108  {
109  std::cout << err.what() << " Dumping execution log **BEGIN**" << std::endl;
110  m_model.stopExecutionTrace();
111 
112  auto trace = m_model.executionTrace();
113  for (auto instr : trace)
114  {
115  instr.printInstructionTrace();
116  }
117  std::cout << "Execution log dump **FINISHED**" << std::endl;
118  throw;
119  }
120  else
121  {
122  throw;
123  }
124  }

161 template <typename T>
162 void PISAProgramRuntime<T>::setParamMemoryToMultiRegisterDeviceMemory(std::pair<std::string, std::vector<T>> trace_memory_map)
163 {
164  try
165  {
166  if (trace_memory_map.second.size() % m_model.getMultiRegisterWidth() != 0)
167  {
168  throw std::runtime_error("Input size not a multiple of multi_register size, mapping undefined");
169  }
170 
171  uint slice_count = trace_memory_map.second.size() / m_model.getMultiRegisterWidth();
172  for (int x = 0; x < slice_count; x++)
173  {
174  pisa::MultiRegister<T> mem_slice(m_model.getMultiRegisterWidth());
175  for (int a = 0; a < m_model.getMultiRegisterWidth(); a++)
176  {
177  mem_slice[a] = trace_memory_map.second[m_model.getMultiRegisterWidth() * x + a];
178  }
179  std::string memory_address = trace_memory_map.first + "_" + std::to_string(x);
180  setMemory(std::pair<std::string, pisa::MultiRegister<T>>(memory_address, mem_slice));
181  }
182  }
183  catch (...)
184  {
185  throw;
186  }

233 template <typename T>
234 std::pair<std::string, std::vector<T>> PISAProgramRuntime<T>::getParamMemoryFromMultiRegisterDeviceMemory(std::string memory_2_param_root)
235 {
236  try
237  {
238  std::pair<std::string, std::vector<T>> return_value;
239  return_value.first = memory_2_param_root;
240 
241  std::vector<std::string> register_names_3_param = m_model.getMatching3ParamRegisterNames(memory_2_param_root);
242  std::vector<std::pair<std::string, int>> indexed_list;
243  int start = memory_2_param_root.length() + 1;
244 
245  for (auto a : register_names_3_param)
246  {
247  auto index = std::stoi(a.substr(start, a.length() - start));
248  indexed_list.push_back(std::pair<std::string, int>(a, index));
249  }
250 
251  std::sort(indexed_list.begin(), indexed_list.end(),
252  [](const std::pair<std::string, int> &lhs, const std::pair<std::string, int> &rhs) {
253  return lhs.second < rhs.second;
254  });
255  std::vector<T> combined_values;
256  for (auto a : indexed_list)
257  {
258  auto memory = m_model.getMemory().readMemory(a.first);
259  for (auto &val : memory.data())
260  {
261  combined_values.push_back(val);
262  }
263  }
264  return_value.second = combined_values;
265  return return_value;
266  }
267  catch (...)
268  {
269  std::cout << "Error while retrieving device memory" << std::endl;
270  throw;
271  }
272 }
